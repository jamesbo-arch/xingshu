## 一、重置测试账号

### 重置真机账号为 guest（id=3，陈建波James）

```bash
node -e "
const mysql = require('mysql2/promise')
mysql.createConnection({ host:'33.tcp.cpolar.top', port:11028, user:'james', password:'751279', database:'xingshu_dev' })
  .then(db => db.query(\"UPDATE users SET identity='guest', avatar_url=NULL, phone=NULL, nickname='微信用户' WHERE id=3\")
  .then(() => { console.log('reset OK'); db.end() }))
"
```

真机侧配合操作：微信 → 该小程序 → 右上角 `···` → 关于 → 清除缓存，或删除小程序重新进入。

---

### 查看所有用户状态

```bash
node -e "
const mysql = require('mysql2/promise')
async function run() {
  const db = await mysql.createConnection({ host:'33.tcp.cpolar.top', port:11028, user:'james', password:'751279', database:'xingshu_dev' })
  const [rows] = await db.query('SELECT id, openid, nickname, identity, phone, avatar_url IS NOT NULL AS has_avatar FROM users ORDER BY id')
  rows.forEach(r => console.log(r.id, r.identity.padEnd(8), r.nickname.padEnd(10), r.openid.slice(0,18)+'...', r.phone||'', r.has_avatar?'[有头像]':''))
  await db.end()
}
run()
"
```

### 清空日记、评论、互动数据（保留用户和标签）

```bash
node -e "
const mysql = require('mysql2/promise')
async function run() {
  const db = await mysql.createConnection({ host:'33.tcp.cpolar.top', port:11028, user:'james', password:'751279', database:'xingshu_dev' })
  await db.query('DELETE FROM diary_tags')
  await db.query('DELETE FROM interactions')
  await db.query('DELETE FROM comments')
  await db.query('DELETE FROM diaries')
  await db.query('UPDATE users SET diary_count=0, like_count=0, fav_count=0, comment_count=0')
  console.log('清空完成')
  await db.end()
}
run()
"
```

## 四、数据库快查

### 查看日记列表

```bash
node -e "
const mysql = require('mysql2/promise')
async function run() {
  const db = await mysql.createConnection({ host:'33.tcp.cpolar.top', port:11028, user:'james', password:'751279', database:'xingshu_dev' })
  const [rows] = await db.query('SELECT d.id, d.title, d.permission, d.status, u.nickname FROM diaries d JOIN users u ON d.user_id=u.id ORDER BY d.id DESC LIMIT 20')
  rows.forEach(r => console.log(r.id, r.permission.padEnd(8), r.status.padEnd(8), r.nickname.padEnd(8), r.title.slice(0,20)))
  await db.end()
}
run()
"
```

## 五、升降级身份

### 升级为 member（按 id）

```bash
node -e "
const mysql = require('mysql2/promise')
mysql.createConnection({ host:'33.tcp.cpolar.top', port:11028, user:'james', password:'751279', database:'xingshu_dev' })
  .then(db => db.query(\"UPDATE users SET identity='member', member_from=CURDATE(), member_until=DATE_ADD(CURDATE(), INTERVAL 365 DAY) WHERE id=3\")
  .then(() => { console.log('升级 member OK'); db.end() }))
"
```
